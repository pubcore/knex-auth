import Knex from 'knex'
import {dbTypes} from '../src/lib/cols'
 
const stampToDate = stamp => {
	var date = stamp.split(/[\s-:]/)
	return (new Date(Date.UTC(date[0], date[1]-1, date[2], date[3], date[4], date[5])))
}
 
const convertDateTime = row => {
	Object.keys(row).forEach(col => {
		if(dbTypes[col] === 'dateTime'){
			row[col] = stampToDate(row[col])
		}
	})
}
 
export default ({table, rows, beforeEach, after}) => {
	var db = new Knex({
		client: 'sqlite3',
		connection: {
			filename: './mydb.sqlite',
			timezone: 'UTC',
		},
		useNullAsDefault:true,
		//sqlite does not create js Date objects for dateTime fields
		postProcessResponse: result => {
			if (Array.isArray(result)) {
				return result.map(row => convertDateTime(row))
			} else {
				return convertDateTime(result)
			}
		}
	})
 
	beforeEach(() => db.schema.dropTableIfExists(table)
		.then(
			() => db.schema.createTable(
				table,
				t => Object.keys(dbTypes).forEach(col => {
					var type = dbTypes[col]
					t[type](col)
				})
			))
		.then(() => Promise.resolve(rows))
		.then(rows => db.batchInsert(table, rows)))
 
	after(() => db.destroy())
	return db
}
